** Load an Utz File                                                :function:

  #+BEGIN_SRC emacs-lisp
    (defun utz/load-utz-file (file)
      (org-babel-load-file
       (expand-file-name (concat file ".org")
			 utz-utz-directory)))
  #+END_SRC

** Edit this Config File                                           :function:

  #+BEGIN_SRC emacs-lisp
    (defun utz/edit-config-file ()
      (interactive)
      (find-file utz-emacs-config-file))
  #+END_SRC

** Reload This Config File                                         :function:

  #+BEGIN_SRC emacs-lisp
    (defun utz/reload-config ()
      (interactive)
      (org-babel-load-file utz-emacs-config-file))
  #+END_SRC

** Narrow / Widen

   #+BEGIN_SRC emacs-lisp
     (defun utz/narrow-page-forward ()
       "Widen, forward, then narrow."
       (interactive)
       (widen)
       (forward-page)
       (narrow-to-page))
   #+END_SRC

   #+BEGIN_SRC emacs-lisp
     (defun utz/narrow-page-backward ()
       "Widen, backward, then narrow."
       (interactive)
       (widen)
       (backward-page)
       (backward-page)
       (narrow-to-page))
   #+END_SRC
