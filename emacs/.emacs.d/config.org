      (cl-defun utz/view-readme (repo-name &key (file-name "README.md") full-path)
	"View the readme file from the repository downloaded by =straight.el=."

*** Cycle Theme Mode

    #+BEGIN_SRC emacs-lisp
      (defun utz/cycle-theme-mode ()
	(interactive)
	(let ((theme-mode (if
			      (equal utz-theme-mode 'light)
			      'dark
			    'light)))
	  (setq utz-theme-mode theme-mode)
	  (load-theme (utz/get-theme-by-mode theme-mode) t)))
    #+END_SRC

*** Next Theme

    #+BEGIN_SRC emacs-lisp
      (defun utz/next-theme ()
	(interactive)
	(let* ((themes (sort (custom-available-themes) 'string<))
	       (current-index (seq-position themes
					    (car custom-enabled-themes)))
	       (next-index (+ current-index 1))
	       (next-theme (seq-elt themes next-index)))
	  (load-theme next-theme)
          (utz/display-current-theme)))
    #+END_SRC

*** Previous Theme

    #+BEGIN_SRC emacs-lisp
      (defun utz/prev-theme ()
	(interactive)
	(let* ((themes (sort
			(custom-available-themes)
			'string<))
	       (current-index (seq-position themes
					    (car custom-enabled-themes)))
	       (prev-index (- current-index 1))
	       (prev-theme (seq-elt themes prev-index)))
	(load-theme prev-theme)
        (utz/display-current-theme)))
    #+END_SRC
