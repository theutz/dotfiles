#+TITLE: My literate emacs config

* Basic Settings  
   
  Herein lies some basic settings that I use as a foundation for my emacsian bliss.

** Text Encoding
    
   Use UTF-8 everywhere
    
   #+BEGIN_SRC emacs-lisp
     (setq-default buffer-file-coding-system 'utf-8-unix)
     (set-default-coding-systems 'utf-8-unix)
     (setq locale-coding-system 'utf-8-unix)
     (prefer-coding-system 'utf-8-unix)
   #+END_SRC

** Exec Path
    
   Make sure that emacs has access to all my homebrewy goodness.
    
   #+BEGIN_SRC emacs-lisp
     (setenv "PATH" (concat (getenv "PATH") ":/usr/local/bin"))
     (setq exec-path (append exec-path '("/usr/local/bin")))
   #+END_SRC
    
** Config files
    
   Setup some constants for my config file paths.
    
   #+BEGIN_SRC emacs-lisp
     (defconst utz-config-init (expand-file-name "init.el" user-emacs-directory))
     (defconst utz-config-file (expand-file-name "literally.org" user-emacs-directory))
     (defconst utz-custom-file custom-file)
     (defconst utz-themes-directory custom-theme-directory)
     (defconst utz-theme-file (expand-file-name "utz-theme.el" utz-themes-directory))
   #+END_SRC
   
   Load the custom file.

   #+BEGIN_SRC emacs-lisp
     (load utz-custom-file 'noerror 'nomessage)
   #+END_SRC
    
** Backup files
    
   #+BEGIN_SRC emacs-lisp
     (setq backup-directory-alist
	   `(("." . ,(concat user-emacs-directory "backups"))))
   #+END_SRC

* Custom Functions

** Narrow Page Navigation
   
   #+BEGIN_SRC emacs-lisp
     (defun utz/narrow-page-forward ()
       "Widen, forward, then narrow."
       (interactive)
       (widen)
       (forward-page)
       (narrow-to-page))

     (defun utz/narrow-page-backward ()
       "Widen, backward, then narrow."
       (interactive)
       (widen)
       (backward-page)
       (backward-page)
       (narrow-to-page))
   #+END_SRC

** Load and Edit Config Files
   
   #+BEGIN_SRC emacs-lisp
     (defun utz/load-config-file ()
       "Load configuration file all over again."
       (interactive)
       (org-babel-load-file utz-config-file))

     (defun utz/edit-config-file ()
       "Edit the configs."
       (interactive)
       (find-file utz-config-file))

     (defun utz/edit-theme-file (x)
       "Edit my custom theme. `C-u` visits it in customize."
       (interactive "p")
       (cond ((eq 4 x) (custom-theme-visit-theme 'utz))
	     (t (find-file utz-theme-file))))
   #+END_SRC

** Open Customize
   
   #+BEGIN_SRC emacs-lisp
     (defun utz/customize ()
       "Open the customize in another window."
       (interactive)
       (evil-window-vsplit)
       (customize))
   #+END_SRC

*** Setup "Narrow / Widen" prefix keys
     
    Narrow, then widen. The circle of lifemacs.
      
    #+BEGIN_SRC emacs-lisp
      (utz/set-leader-key :infix "n"
	"" '(:ignore t :wk "Narrow / Widen")
	"." '(utz/hydra-narrow/body :wk "Transient State")
	"w" '(widen :wk "Widen")
	"p" '(narrow-to-page :wk "Narrow to Page")
	"r" '(narrow-to-region :wk "Narrow to Region")
	"d" '(narrow-to-defun :wk "Narrow to Function Definition")
	"]" '(utz/narrow-page-forward wk: "Next Page")
	"[" '(utz/narrow-page-backward wk: "Previous Page"))
    #+END_SRC

    Here's the hydra for the transient state:

    #+BEGIN_SRC emacs-lisp
      (defhydra utz/hydra-narrow (:hint nil)
	"
	^
	^Narrow^              ^Widen^    ^Navigate^
	^------^--------------^-----^----^--------^---------
	_p_ narrow to page    _w_ widen  _]_ next page
	_d_ narrow to defun   ^ ^        _[_ previous page
	_r_ narrow to region  ^ ^        ^ ^
	^ ^                   ^ ^        ^ ^
	"
	("q" nil "cancel")
	("p" narrow-to-page)
	("d" narrow-to-defun)
	("r" narow-to-region)
	("w" widen)
	("]" utz/narrow-page-forward)
	("[" utz/narrow-page-backward))
    #+END_SRC
